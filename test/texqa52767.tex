%% \c@mv@.... の値を \count18+1 にセットする マクロ
\def\setMVNtoSYN{%
  \begingroup
  \advance\count18\@ne
  \def\version@elt##1{%
    \csname c@\expandafter\@gobble\string##1\endcsname=\number\count18\relax
  }%
  \global\version@list
  \endgroup
}
